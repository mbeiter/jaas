33  package org.beiter.michael.authn.jaas.common;
34  
35  /**
36   * The events that are available in a JAAS module.
37   */
38  public enum Events {
39      //CHECKSTYLE:OFF
40      AUTHN_ATTEMPT("A user's credentials have been successfully validated"),
41      AUTHN_FAILURE("A user's credentials could not be validated"),
42      AUTHN_ERROR("A service failure prevented the system from completing the request"),
43      AUTHN_COMMIT_FAILURE("A commit was aborted due to earlier failures"),
44      AUTHN_ABORT_FAILURE("A login attempt was aborted due to earlier credential validation failures"),
45      AUTHN_ABORT_COMMIT("Although the authentication succeeded, a login attempt was aborted due to the commit failing"),
46      AUTHN_ABORT_CHAIN("Although the authentication and the commit succeeded, a login attempt was aborted due to the commit of another module failing"),
47      AUTHN_SUCCESS("A user has been successfully authenticated"),
48      AUTHN_LOGOUT("A user has been logged out");
49  
50      private final String value;
51  
52      private Events(final String keyName) {
53          this.value = keyName;
54      }
55  
56  
57      public String getValue() {
58          return value;
59      }
60      //CHECKSTYLE:ON
61  }
