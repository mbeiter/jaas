39  import org.beiter.michael.authn.jaas.loginmodules.password.common.PasswordAuthenticator;

43  import org.junit.Test;

45  import org.slf4j.LoggerFactory;

47  import static org.hamcrest.Matchers.*;
48  import static org.junit.Assert.assertThat;

50  public class PasswordAuthenticatorFactoryExtendedTest {
51  
52      /**
53       * The logger object for this class
54       */
55      private static final Logger LOG = LoggerFactory.getLogger(PasswordAuthenticatorFactoryExtendedTest.class);
56  
57      /**
58       * The test class to instantiate
59       */
60      private static final String className = "org.beiter.michael.authn.jaas.loginmodules.password.authenticators.dummy.DummyPasswordAuthenticator";
61  
62      /**
63       * Reset the password authenticator factory to allow creating several instances of the underlying password
64       * authenticator implementations.
65       */
66      @Before
67      public void unsetSingletonInFactory() {
68  
69          PasswordAuthenticatorFactory.reset();
70      }
71  
72      /**
73       * Retrieve a specific implementation of the PasswordAuthenticator interface, and assert that the returned
74       * implementation equals the requested implementation.
75       */
76      @Test
77      public void getSpecificImplementationTest() {
78  
79          CommonProperties commonProps = JaasBasedCommonPropsBuilder.buildDefault();
80  
81          PasswordAuthenticator passwordAuthenticator;
82          try {
83              passwordAuthenticator = PasswordAuthenticatorFactory.getInstance(className, commonProps);
84          } catch (FactoryException e) {
85              AssertionError ae = new AssertionError("Instantiation error");
86              ae.initCause(e);
87              throw ae;
88          }
89  
90          String error = "The class instantiated by the factory does not match the expected class";
91          assertThat(error, DummyPasswordAuthenticator.class.getCanonicalName(), is(equalTo(className)));
92          assertThat(error, passwordAuthenticator.getClass().getCanonicalName(), is(equalTo(className)));
93      }
94  
95      /**
96       * Retrieve two instances of a specific implementation of the PasswordAuthenticator interface, and asserts that the
97       * returned objects are two separate instances.
98       */
99      @Test
100     public void twoInstancesAreDifferentTest() {
101 
102         CommonProperties commonProps = JaasBasedCommonPropsBuilder.buildDefault();
103 
104         PasswordAuthenticator passwordAuthenticator1, passwordAuthenticator2;
105         try {
106             passwordAuthenticator1 = PasswordAuthenticatorFactory.getInstance(className, commonProps);
107             passwordAuthenticator2 = PasswordAuthenticatorFactory.getInstance(className, commonProps);
108         } catch (FactoryException e) {
109             AssertionError ae = new AssertionError("Instantiation error");
110             ae.initCause(e);
111             throw ae;
112         }
113 
114         String error = "The factory returns a singleton instead of a new object";
115         assertThat(error, passwordAuthenticator1, is(not(sameInstance(passwordAuthenticator2))));
116     }
117 
118     /**
119      * Retrieve two singleton instances of a specific implementation of the PasswordAuthenticator interface, and asserts
120      * that the two returned objects are identical (i.e. the factory returns a singleton).
121      * <p>
122      * Then, a regular (non-singleton) instance is retrieved, which are asserted to be different than the previously
123      * retrieved objects.
124      * <p>
125      * Finally, the factory is reset, and another instance is retrieved. If the factory resets properly, the third
126      * instance must be unequal to the first three instances.
127      */
128     @Test
129     public void factoryReturnsSingletonTest() {
130 
131         CommonProperties commonProps = JaasBasedCommonPropsBuilder.buildDefault();
132 
133         // test that two singletons retrieved from the factory are identical
134         PasswordAuthenticator passwordAuthenticator1, passwordAuthenticator2;
135         try {
136             passwordAuthenticator1 = PasswordAuthenticatorFactory.getSingleton(className, commonProps);
137             passwordAuthenticator2 = PasswordAuthenticatorFactory.getSingleton(className, commonProps);
138         } catch (FactoryException e) {
139             AssertionError ae = new AssertionError("Instantiation error");
140             ae.initCause(e);
141             throw ae;
142         }
143 
144         String error = "The factory does not return a singleton";
145         assertThat(error, passwordAuthenticator1, is(sameInstance(passwordAuthenticator2)));
146 
147         // then test that a regular (non-singleton) instance is different
148         PasswordAuthenticator passwordAuthenticator3;
149         try {
150             passwordAuthenticator3 = PasswordAuthenticatorFactory.getInstance(className, commonProps);
151         } catch (FactoryException e) {
152             AssertionError ae = new AssertionError("Instantiation error");
153             ae.initCause(e);
154             throw ae;
155         }
156         error = "The factory returns a singleton instead of a new object";
157         assertThat(error, passwordAuthenticator1, is(not(sameInstance(passwordAuthenticator3))));
158         assertThat(error, passwordAuthenticator2, is(not(sameInstance(passwordAuthenticator3))));
159 
160         // reset the factory
161         PasswordAuthenticatorFactory.reset();
162 
163         // now test that the factory return a new object (i.e. a new singleton)
164         PasswordAuthenticator passwordAuthenticator4;
165         try {
166             passwordAuthenticator4 = PasswordAuthenticatorFactory.getSingleton(className, commonProps);
167         } catch (FactoryException e) {
168             AssertionError ae = new AssertionError("Instantiation error");
169             ae.initCause(e);
170             throw ae;
171         }
172 
173         error = "The factory does not return a singleton, or does not reset properly";
174         assertThat(error, passwordAuthenticator1, is(not(sameInstance(passwordAuthenticator4))));
175         assertThat(error, passwordAuthenticator2, is(not(sameInstance(passwordAuthenticator4))));
176         assertThat(error, passwordAuthenticator3, is(not(sameInstance(passwordAuthenticator4))));
177     }
