62   * This authenticator performs validation of username / password type credentials against JDBC databases.
63   * <p>
64   * While any type of JDBC database should be supported, this class has only been tested with MySQL.

111     /**
112      * Authenticate a user by validating the user's password, returning a Subject with one or more {@code Principal}s
113      * set (if validation was successful), or throw a {@code LoginException} (if validation fails)
114      * <p>
115      * This method connects to the database as configured in the the configuration provided during initialization, and
116      * uses the configured parameterized SQL query to retrieve a user's ID and credential using the provided userName
117      * and domain to locate them in the database. If initialization has not been completed, the authentication fails.
118      * <p>
119      * The configured SQL query must take two query parameters, the first being the domain, the second being the
120      * userName. If the domain is not used in the database, the SQL query must be crafted so that the first parameter
121      * is irrelevant.
122      * <p>
123      * The configured SQL query must return two columns, the first being the user's ID, the second being the credential
124      * against which the provided password is to be validated.
125      * <p>
126      * If the JDBC call failed, no record could be found, the parameters do not match, or similar SQL issues ocured,
127      * a {@link javax.security.auth.login.LoginException} is thrown.
128      * <p>
129      * The provided password is validated against the credential using the parent class' {@code passwordValidator}
130      * object.
131      * <p>
132      * If the validation fails (i.e. the provided password does not match the credential), a
133      * {@link javax.security.auth.login.FailedLoginException} is thrown.
134      * <p>
135      * If the validation is successful, a {@code Subject} is populated with three principals is returned: The
136      * user's ID as returned by the database, and both the user provided domain and the user provided principal (i.e.
137      * the the identifiers used to authenticate the users).
138      *
139      * @param domain            The white label domain in which the username is located
140      * @param userName          The username to authenticate with
141      * @param password          The password to authenticate with
142      * @param passwordValidator The validator to use for the password
143      * @return a callback for querying subject attributes.
144      * @throws LoginException when this {@code LoginModule} is unable to perform the authentication. Catch a
145      *                        {@link javax.security.auth.login.FailedLoginException} to determine if the authentication
146      *                        failed due to an incorrect password
147      */
148     // The logical flow in this method is not overly complex, the warning results from parameter validation
149     // CHECKSTYLE:OFF
150     @SuppressWarnings({"PMD.CyclomaticComplexity", "PMD.NPathComplexity"})
151     // CHECKSTYLE:ON
152     public final Subject authenticate(final String domain, final String userName, final char[] password,
153                                       final PasswordValidator passwordValidator)
154             throws LoginException {
155 
156         // make sure the credentials are not null
157         if (domain == null || userName == null || password == null) {
158             throw new LoginException("The credentials cannot be null");
159         }
160 
161         // SQL query is required
162         if (StringUtils.isBlank(dbProps.get().getSqlUserQuery())) {
163             final String error = "Invalid SQL user authentication query (query is null or empty)";
164             LOG.warn(error);
165             throw new LoginException(error);
166         }
167 
168         final UserRecord userRecord = getUserRecord(domain, userName);
169 
170         if (userRecord.getUserId() == null || userRecord.getUserId().length() == 0) {
171             final String error = "User ID for username '" + userName + "' is null or empty in the database";
172             LOG.warn(error);
173             throw new LoginException(error);
174         }
175 
176         if (userRecord.getCredential() == null || userRecord.getCredential().length() == 0) {
177             final String error = "Credential for username '" + userName + "' / user ID '" + userRecord.getUserId()
178                     + "' is null or empty in the database";
179             LOG.warn(error);
180             throw new LoginException(error);
181         }
182 
183         // no need for defensive copies of Strings, but create a defensive copy of the password
184         final char[] myPassword = password.clone();
185 
186         // convert the credential string to a char array
187         final char[] myCredential = userRecord.getCredential().toCharArray();
188 
189         if (!passwordValidator.validate(myPassword, myCredential)) {
190             final String error = "Invalid password for username '" + userName + "'";
191             LOG.info(error);
192             throw new FailedLoginException(error);
193         }
194 
195         // The authentication was successful!
196         // Create the subject and clean up confidential data as far as possible.
197 
198         // clear the char representation of the credential
199         Cleanser.wipe(myCredential);
200 
201         // clear the defensive copy of the password created earlier
202         Cleanser.wipe(myPassword);
203 
204         // create a principal that includes the username and domain name that were used to authenticate the user
205         final UserPrincipal userPrincipal = new UserPrincipal(userRecord.getUserId(), domain, userName);
206 
207         // wrap the principal in a Subject
208         final Subject subject = new Subject();
209         subject.getPrincipals().add(userPrincipal);
210 
211         return subject;
212     }

214     /**
215      * Retrieve a user record from the database, with the user record being uniquely identified through {@code domain}
216      * and {@code userName}.
217      *
218      * @param domain   The domain in which the {@code userName} is located
219      * @param userName The username to search for
220      * @return A user record containing domain, username, user ID (from the DB), and credentials (from the DB)
221      * @throws LoginException When the JDBC cvonnection failed, or the username / domain combination could not be found
222      */
223     // The SQL statement is retrieved from the configuration, and the admin is trusted
224     @edu.umd.cs.findbugs.annotations.SuppressWarnings("SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING")
225     private UserRecord getUserRecord(final String domain, final String userName)
226             throws LoginException {
227 
228         String userId;
229         String credential;
230 
231         Connection connection = null;
232         PreparedStatement statement = null;
233         ResultSet resultSet = null;
234 
235         try {
236             connection = getDatabaseConnection();
237 
238             statement = connection.prepareStatement(dbProps.get().getSqlUserQuery());
239             statement.setString(1, domain);
240             statement.setString(2, userName);
241 
242             resultSet = statement.executeQuery();
243 
244             if (resultSet.next()) {
245                 userId = resultSet.getString(1);
246                 credential = resultSet.getString(2);
247             } else {
248                 final String error = "Username '" + userName + "' does not exist (query returned zero results)";
249                 LOG.warn(error);
250                 throw new LoginException(error);
251             }
252 
253             resultSet.close();
254             statement.close();
255         } catch (SQLException e) {
256             final String error = "Error executing SQL query";
257             LOG.warn(error, e);
258             throw Util.newLoginException(error, e);
259         } finally {
260             DbUtil.close(resultSet);
261             DbUtil.close(statement);
262             DbUtil.close(connection);
263         }
264 
265         return new UserRecord(domain, userName, userId, credential);
266     }

268     /**
269      * Obtain a database connection - either a JNDI connection (directly from the factory), or a pooled JDBC connection
270      *
271      * @return a pooled database connection
272      * @throws LoginException when the connection cannot be retrieved from JNDI or the connection pool, or the pool
273      *                        cannot be created
274      */
275     private Connection getDatabaseConnection()
276             throws LoginException {
277 
278         Connection connection;
279         if (StringUtils.isNotEmpty(dbProps.get().getJndiConnectionName())) {
280             try {
281                 connection = ConnectionFactory.getConnection(dbProps.get().getJndiConnectionName());
282             } catch (FactoryException e) {
283                 final String error = "Could not retrieve JNDI database connection";
284                 LOG.warn(error, e);
285                 throw Util.newLoginException(error, e);
286             }
287         } else {
288             try {
289                 // connection spec is required
290                 if (connProps.get() == null) {
291                     final String error = "Database connection pool configuration has not been provided or initialized";
292                     LOG.warn(error);
293                     throw new FactoryException(error);
294                 }
295 
296                 // driver is required
297                 if (StringUtils.isBlank(connProps.get().getDriver())) {
298                     final String error = "Invalid database driver (driver name is null or empty)";
299                     LOG.warn(error);
300                     throw new FactoryException(error);
301                 }
302 
303                 // url is required
304                 if (StringUtils.isBlank(connProps.get().getUrl())) {
305                     final String error = "Invalid database URL (URL is null or empty)";
306                     LOG.warn(error);
307                     throw new FactoryException(error);
308                 }
309 
310                 connection = ConnectionFactory.getConnection(connProps.get());
311 
312             } catch (FactoryException e) {
313                 final String error = "Could not create pooled database connection";
314                 LOG.warn(error, e);
315                 throw Util.newLoginException(error, e);
316             }
317         }
318 
319         return connection;
320     }
