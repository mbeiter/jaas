33  package org.beiter.michael.authn.jaas.common.authenticator;
34  
35  import org.beiter.michael.authn.jaas.common.validator.PasswordValidator;
36  
37  import javax.security.auth.Subject;
38  import javax.security.auth.login.LoginException;
39  import java.util.Map;
40  
41  /**
42   * An interface representing a username / password based authentication mechanism.
43   * <p/>
44   * Classes implementing this interface <b>must</b> be thread safe.
45   */
46  public interface PasswordAuthenticator {
47  
48      /**
49       * Initializes the authenticator configuration
50       * <p/>
51       * A class implementing this interface must provide a reasonable default configuration and handle situations where
52       * the {@code validate()} method is called without a previous call of {@code init()} (i.e. do not throw a runtime
53       * exception).
54       * <p/>
55       * A class implementing this interface must ensure that subsequent calls to this method update the class'
56       * configuration in a thread-safe way.
57       *
58       * @param properties The properties to initialize the message queue with. Supported parameters may vary with the
59       *                   implementing classes.
60       */
61      void init(Map<String, ?> properties);
62  
63      /**
64       * Authenticate a user by validating the user's password, returning a Subject with one or more {@code Principal}s
65       * set (if validation was successful), or throw a {@code LoginException} (if validation fails)
66       *
67       * @param domain            The white label domain in which the username is located
68       * @param userName          The username to authenticate with
69       * @param password          The password to authenticate with
70       * @param passwordValidator The validator to use for the password
71       * @return a callback for querying subject attributes.
72       * @throws LoginException when this {@code LoginModule} is unable to perform the authentication. Catch a
73       *                        {@link javax.security.auth.login.FailedLoginException} to determine if the authentication
74       *                        failed due to an incorrect password
75       */
76      Subject authenticate(String domain, String userName, char[] password, PasswordValidator passwordValidator)
77              throws LoginException;
78  
79  }
