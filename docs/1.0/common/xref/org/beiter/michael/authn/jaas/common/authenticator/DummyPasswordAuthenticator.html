33  package org.beiter.michael.authn.jaas.common.authenticator;
34  
35  import org.beiter.michael.authn.jaas.common.UserPrincipal;
36  import org.beiter.michael.authn.jaas.common.Util;
37  import org.beiter.michael.authn.jaas.common.validator.PasswordValidator;
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  
41  import javax.security.auth.Subject;
42  import javax.security.auth.login.FailedLoginException;
43  import javax.security.auth.login.LoginException;
44  import java.util.Map;
45  
46  /**
47   * This authenticator performs validation of username / password type credentials by checking whether the provided
48   * username matches the provided password. This is obviously not a very strong authenticator, and is intended to be
49   * used for testing only.
50   */
51  public class DummyPasswordAuthenticator
52          implements PasswordAuthenticator {
53  
54      /**
55       * The logger object for this class
56       */
57      private static final Logger LOG = LoggerFactory.getLogger(DummyPasswordAuthenticator.class);
58  
59  
60      /**
61       * {@inheritDoc}
62       */
63      @Override
64      public final void init(final Map<String, ?> properties) {
65          // nothing to do here - this implementation does not take any properties
66      }
67  
68      /**
69       * Authenticate a user by validating the user's password, returning a Subject with one or more {@code Principal}s
70       * set (if validation was successful), or throw a {@code LoginException} (if validation fails)
71       * <p/>
72       * This method checks whether the provided username matches the provided password. This requires a suitable plain
73       * text password validator to work, as the actual validation is performed by the provided
74       * {@code PasswordValidator}). The method also checks that the domain is not {@code null} (but it does allow an
75       * empty String value).
76       * <p/>
77       * If the validation is successful, a {@code Subject} is populated with three principals is returned: The
78       * user's ID as a concatenation of the username with the string "ID:", and both the user provided domain and the
79       * user provided principal (i.e. the the identifiers used to authenticate the users).
80       *
81       * @param domain            The white label domain in which the username is located
82       * @param userName          The username to authenticate with
83       * @param password          The password to authenticate with
84       * @param passwordValidator The validator to use for the password
85       * @return a callback for querying subject attributes.
86       * @throws LoginException when this {@code LoginModule} is unable to perform the authentication. Catch a
87       *                        {@link javax.security.auth.login.FailedLoginException} to determine if the authentication
88       *                        failed due to an incorrect password
89       */
90      public final Subject authenticate(final String domain, final String userName, final char[] password,
91                                        final PasswordValidator passwordValidator)
92              throws LoginException {
93  
94          // make sure the credentials are not null
95          if (domain == null) {
96              throw new LoginException("The domain cannot be null");
97          }
98          if (userName == null) {
99              throw new LoginException("The username cannot be null");
100         }
101         if (password == null) {
102             throw new LoginException("The password cannot be null");
103         }
104 
105         // no need for defensive copies of Strings, but create a defensive copy of the password
106         final char[] myPassword = password.clone();
107 
108         // convert the username string to a char array
109         final char[] myCredential = userName.toCharArray();
110 
111         // perform the password validation
112         if (!passwordValidator.validate(myPassword, myCredential)) {
113             final String error = "Invalid password for username '" + userName + "'";
114             LOG.info(error);
115             throw new FailedLoginException(error);
116         }
117 
118         // The authentication was successful!
119         // Create the subject and clean up confidential data as far as possible.
120 
121         // clear the char representation of the credential
122         Util.zeroArray(myCredential);
123 
124         // clear the defensive copy of the password created earlier
125         Util.zeroArray(myPassword);
126 
127         // create a principal that includes the username and domain name that were used to authenticate the user
128         final UserPrincipal userPrincipal = new UserPrincipal("ID:" + userName, domain, userName);
129 
130         // wrap the principal in a Subject
131         final Subject subject = new Subject();
132         subject.getPrincipals().add(userPrincipal);
133 
134         return subject;
135     }
136 }
