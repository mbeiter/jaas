33  package org.beiter.michael.authn.jaas.authenticator.jdbc;
34  
35  import org.beiter.michael.db.ConnectionFactory;
36  import org.beiter.michael.db.ConnectionPoolSpec;
37  import org.beiter.michael.db.ConnectionSpec;
38  import org.beiter.michael.db.FactoryException;
39  import org.h2.jdbcx.JdbcConnectionPool;
40  import org.h2.tools.Server;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  import java.sql.Connection;
45  import java.sql.ResultSet;
46  import java.sql.SQLException;
47  import java.util.concurrent.atomic.AtomicBoolean;
48  
49  /**
50   * A utility class that starts an H2 database server and populates it with test data for username / password validation
51   * to test the JDBC database connection and JAAS module.
52   */
53  public class H2Server {
54  
55      private static final Logger LOG = LoggerFactory.getLogger(H2Server.class);
56  
57      public static final String DRIVER = "org.h2.Driver";
58      public static final String DATABASE = "h2test_db";
59      public static final String URL = "jdbc:h2:mem:" + DATABASE;
60      public static final String USER = "someUser";
61      public static final String PASSWORD = "noPassword";
62  
63      private static AtomicBoolean serverRunning = new AtomicBoolean(false);
64      private static volatile Server server;
65  
66      // prevent instantiation of this class
67      private H2Server() {
68      }
69  
70      /**
71       * Initializes the default schema and sets default values (if feasible). This deletes any data that are already in
72       * the database!
73       *
74       * @throws FactoryException When there is a problem getting a connection to the database
75       * @throws SQLException     When there is a problem initializing the database. Note that this can leave the database in
76       *                          an inconsistent state.
77       */
78      public static synchronized void init()
79              throws FactoryException, SQLException {
80  
81          LOG.info("Initializing the in-memory database server with a default schema and default values");
82  
83          ConnectionSpec connSpec = new ConnectionSpec(URL, USER, PASSWORD);
84          ConnectionPoolSpec poolSpec = new ConnectionPoolSpec();
85  
86          Connection con = ConnectionFactory.getConnection(DRIVER, connSpec, poolSpec);
87  
88          String stmt = "DROP TABLE IF EXISTS user_plaintext";
89          con.prepareStatement(stmt).execute();
90  
91          // a user table that stores the password in plain text together with the username
92          stmt = "CREATE TABLE user_plaintext ("
93                  + " id INT NOT NULL PRIMARY KEY,"
94                  + " domain VARCHAR(255) NOT NULL,"
95                  + " username VARCHAR(255) NOT NULL,"
96                  + " password VARCHAR(255)"
97                  + ")";
98          con.prepareStatement(stmt).execute();
99  
100         stmt = "INSERT INTO user_plaintext (id, domain, username, password) VALUES "
101                 + "  (11, 'domain1', 'user1', 'user1')"
102                 + ", (12, 'domain1', 'user2', '')"
103                 + ", (13, 'domain1', 'user3', 'password3')"
104                 + ", (21, 'domain2', 'user1', 'password1')"
105                 + ", (22, 'domain2', 'user2', 'password2')";
106         con.prepareStatement(stmt).execute();
107 
108         // check that the user insert was successful
109         stmt = "SELECT COUNT(id) AS count FROM user_plaintext";
110         ResultSet rs = con.prepareStatement(stmt).executeQuery();
111 
112         if (rs.next()) {
113             if (rs.getInt("count") != 5) {
114                 String error = "Result set row count mismatch: expected 5, found " + rs.getInt("count");
115                 LOG.warn(error);
116                 throw new IllegalStateException(error);
117             }
118         } else {
119             String error = "The result set is empty";
120             LOG.warn(error);
121             throw new IllegalStateException(error);
122         }
123 
124         con.close();
125     }
126 
127     /**
128      * Starts the server, but does not perform any database schema initialization
129      *
130      * @throws SQLException When the server cannot be started
131      */
132     public static void start()
133             throws SQLException {
134 
135         LOG.info("Checking if the in-memory database server can be started");
136 
137         // make sure the URL type and database type match
138         if (!URL.startsWith("jdbc:h2:mem:")) {
139             final String error = "Not an H2 connection URL: " + URL;
140             LOG.error(error);
141             throw new IllegalStateException(error);
142         }
143 
144         // load the driver
145         try {
146             Class.forName(DRIVER);
147         } catch (ClassNotFoundException e) {
148             // a SQL exception is not the ideal thing to throw here, but it is good enough for a testing setup like this
149             String error = "Driver class not found, cannot start the DB server: " + DRIVER;
150             LOG.error(error, e);
151             throw new SQLException(error, e);
152         }
153 
154         // start the server only if there is not already a server running
155         if (serverRunning.compareAndSet(false, true)) {
156 
157             try {
158                 LOG.info("Starting the in-memory database server");
159                 server = Server.createTcpServer();
160                 server.start();
161                 LOG.info("In-memory database server started");
162             } catch (SQLException e) {
163                 serverRunning.set(false);
164                 LOG.warn("In-memory database server not started dues to error");
165                 throw new SQLException(e);
166             }
167         } else {
168             LOG.info("In-memory database server already running");
169         }
170 
171 
172         // connect to the DB (using the internal H2 connection pool) to trigger creation of an empty database owned by
173         // the correct user. The H2 documentation says that even if we throw the pool away immediately, using the pool
174         // is still faster than getting a direct connection with the driver manager...
175         JdbcConnectionPool cp = JdbcConnectionPool.create(URL, USER, PASSWORD);
176         Connection conn = cp.getConnection();
177         conn.close();
178         cp.dispose();
179     }
180 
181     /**
182      * Stops the server
183      */
184     public static void stop() {
185 
186         LOG.info("Checking if the in-memory database server can be stopped");
187 
188         // stop the server only if there is a server running
189         if (serverRunning.compareAndSet(true, false)) {
190 
191             LOG.info("Stopping the in-memory database server");
192             server.stop();
193             LOG.info("In-memory database server stopped");
194         } else
195             LOG.info("In-memory database server not running");
196     }
197 
198     /**
199      * Restarts the server, but does not perform any database schema initialization
200      *
201      * @throws SQLException When the server cannot be started
202      */
203     public static void restart()
204             throws SQLException {
205 
206         stop();
207         start();
208     }
209 }
