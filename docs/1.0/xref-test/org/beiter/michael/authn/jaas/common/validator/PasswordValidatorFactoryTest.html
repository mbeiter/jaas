33  package org.beiter.michael.authn.jaas.common.validator;

35  import org.beiter.michael.authn.jaas.common.FactoryException;
36  import org.junit.Before;
37  import org.junit.Test;
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;

41  import java.util.concurrent.ConcurrentHashMap;
42  
43  import static org.hamcrest.Matchers.*;
44  import static org.junit.Assert.assertThat;
45  
46  public class PasswordValidatorFactoryTest {
47  
48      private static final Logger log = LoggerFactory.getLogger(PasswordValidatorFactoryTest.class);
49  
50      /**
51       * Reset the password validator factory to allow creating several instances of the underlying password validator
52       * implementations.
53       */
54      @Before
55      public void unsetSingletonInFactory() {
56  
57          PasswordValidatorFactory.reset();
58      }
59  
60      /**
61       * Retrieve the default implementation of the PasswordValidator interface, and asserts that this default
62       * implementation equals the implementation shipped with this library.
63       */
64      @Test
65      public void getDefaultImplementationTest() {
66  
67          PasswordValidator passwordValidator;
68          try {
69              passwordValidator = PasswordValidatorFactory.getInstance(new ConcurrentHashMap<String, Object>());
70          } catch (FactoryException e) {
71              AssertionError ae = new AssertionError("Instantiation error");
72              ae.initCause(e);
73              throw ae;
74          }
75  
76          String error = "The class instantiated by the factory does not match the expected class";
77          assertThat(error, PlainTextPasswordValidator.class.isInstance(passwordValidator), is(equalTo(true)));
78      }
79  
80      /**
81       * Retrieve a specific implementation of the PasswordValidator interface, and asserts that the returned
82       * implementation equals the requested implementation.
83       */
84      @Test
85      public void getSpecificImplementationTest() {
86  
87          String className = "org.beiter.michael.authn.jaas.common.validator.PlainTextPasswordValidator";
88  
89          PasswordValidator passwordValidator;
90          try {
91              passwordValidator = PasswordValidatorFactory.getInstance(className, new ConcurrentHashMap<String, Object>());
92          } catch (FactoryException e) {
93              AssertionError ae = new AssertionError("Instantiation error");
94              ae.initCause(e);
95              throw ae;
96          }
97  
98          String error = "The class instantiated by the factory does not match the expected class";
99          assertThat(error, PlainTextPasswordValidator.class.getCanonicalName(), is(equalTo(className)));
100     }
101 
102     /**
103      * A non-existing class name (i.e. a class not in the class path) should throw an exception
104      */
105     @Test(expected = FactoryException.class)
106     public void getNonExistingImplementationTest()
107             throws FactoryException {
108 
109         PasswordValidatorFactory.getInstance("someGarbageName", new ConcurrentHashMap<String, Object>());
110     }
111 
112     /**
113      * An invalid class name (i.e. a class of the wrong type) should throw an exception
114      */
115     @Test(expected = FactoryException.class)
116     public void getInvalidImplementationTest()
117             throws FactoryException {
118 
119         PasswordValidatorFactory.getInstance(String.class.getCanonicalName(), new ConcurrentHashMap<String, Object>());
120     }
121 
122     /**
123      * Retrieve two instances of the default implementation of the PasswordValidator interface, and asserts that the
124      * two returned objects are identical (i.e. the factory returns a singleton).
125      * <p/>
126      * Then the factory is reset, and another instance is retrieved. If the factory resets properly, the third instance
127      * must be unequal to the first two instances.
128      */
129     @Test
130     public void factoryReturnsSingletonTest() {
131 
132         PasswordValidator passwordValidator1, passwordValidator2;
133         try {
134             passwordValidator1 = PasswordValidatorFactory.getInstance(new ConcurrentHashMap<String, Object>());
135             passwordValidator2 = PasswordValidatorFactory.getInstance(new ConcurrentHashMap<String, Object>());
136         } catch (FactoryException e) {
137             AssertionError ae = new AssertionError("Instantiation error");
138             ae.initCause(e);
139             throw ae;
140         }
141 
142         String error = "The factory does not return a singleton";
143         assertThat(error, passwordValidator1, is(equalTo(passwordValidator2)));
144 
145         // reset the factory
146         PasswordValidatorFactory.reset();
147 
148         // now test that the factory return a new object (i.e. a new singleton)
149         PasswordValidator passwordValidator3;
150         try {
151             passwordValidator3 = PasswordValidatorFactory.getInstance(new ConcurrentHashMap<String, Object>());
152         } catch (FactoryException e) {
153             AssertionError ae = new AssertionError("Instantiation error");
154             ae.initCause(e);
155             throw ae;
156         }
157 
158         error = "The factory does not return a singleton, or does not reset properly";
159         assertThat(error, passwordValidator1, is(not(equalTo(passwordValidator3))));
160     }
