33  package org.beiter.michael.authn.jaas.authenticator.jdbc;
34  
35  import org.beiter.michael.authn.jaas.common.ConfigOptions;
36  
37  /**
38   * The JDBC configuration options used by this library.
39   */
40  public enum JdbcConfigOptions
41          implements ConfigOptions {
42      //CHECKSTYLE:OFF
43  
44      // Database connection config parameters
45      JNDI_NAME("jaas.jdbc.jndi.name"),
46      POOL_DRIVER("jaas.jdbc.pool.driver"),
47      POOL_URL("jaas.jdbc.pool.url"),
48      POOL_USER("jaas.jdbc.pool.user"),
49      POOL_PASSWORD("jaas.jdbc.pool.password"),
50      POOL_MAXTOTAL("jaas.jdbc.pool.maxTotal"),
51      POOL_MAXIDLE("jaas.jdbc.pool.maxIdle"),
52      POOL_MINIDLE("jaas.jdbc.pool.minIdle"),
53      POOL_MAXWAITMILLIS("jaas.jdbc.pool.maxWaitMillis"),
54      POOL_TESTONCREATE("jaas.jdbc.pool.testOnCreate"),
55      POOL_TESTONBORROW("jaas.jdbc.pool.testOnBorrow"),
56      POOL_TESTONRETURN("jaas.jdbc.pool.testOnReturn"),
57      POOL_TESTWHILEIDLE("jaas.jdbc.pool.testWhileIdle"),
58      POOL_TIMEBETWEENEVICTIONRUNS("jaas.jdbc.pool.timeBetweenEvictionRuns"),
59      POOL_NUMTESTSPEREVICTIONRUN("jaas.jdbc.pool.numTestsPerEvictionRun"),
60      POOL_ISLIFO("jaas.jdbc.pool.isLIFO"),
61      POOL_AUTOCOMMIT("jaas.jdbc.pool.autoCommit"),
62      POOL_READONLY("jaas.jdbc.pool.readOnly"),
63      POOL_TRANSACTIONISOLATION("jaas.jdbc.pool.transactionIsolation"),
64      POOL_VALIDATIONQUERY("jaas.jdbc.pool.validationQuery"),
65      POOL_MAXCONNLIFETIME("jaas.jdbc.pool.maxConnLifetime"),
66      // Other config parameters
67      SQL_USERQUERY("jaas.jdbc.sql.userQuery");
68  
69      private final String name;
70  
71      private JdbcConfigOptions(final String keyName) { this.name = keyName; }
72  
73      public String getName() { return name; }
74      //CHECKSTYLE:ON
75  }
