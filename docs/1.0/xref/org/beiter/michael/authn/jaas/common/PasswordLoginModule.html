134     @Override
135     public final void initialize(final Subject subject, final CallbackHandler callbackHandler,
136                                  final Map<String, ?> sharedState, final Map<String, ?> options) {
137 
138         LOG.debug("Initializing");
139 
140         Validate.notNull(subject);
141         Validate.notNull(callbackHandler);
142         Validate.notNull(sharedState);
143         Validate.notNull(options);
144 
145         // keep a reference to the originally provided arguments (no defensive copy)
146         this.pSubject = subject;
147         this.pCallbackHandler = callbackHandler;
148         this.pOptions = options;
149 
150         // initialize the audit object
151         try {
152             final String auditClassName = getOption(JaasConfigOptions.AUDIT_CLASS);
153             if (auditClassName == null) {
154                 LOG.debug("Requesting default audit class from the audit factory");
155                 this.audit = AuditFactory.getInstance(options);
156             } else {
157                 LOG.debug("Requesting audit class instance of '" + auditClassName + "' from the audit factory");
158                 this.audit = AuditFactory.getInstance(auditClassName, options);
159             }
160         } catch (FactoryException e) {
161             final String error = "The audit class cannot be instantiated. This is most likely a configuration"
162                     + " problem. Is the configured class available in the classpath?";
163             LOG.error(error, e);
164             throw new IllegalStateException(error, e);
165         }
166 
167         // initialize the message object
168         try {
169             final String messageClassName = getOption(JaasConfigOptions.MESSAGEQ_CLASS);
170             if (messageClassName == null) {
171                 LOG.debug("Requesting default message class from the message factory");
172                 this.messageQ = MessageQFactory.getInstance(options);
173             } else {
174                 LOG.debug("Requesting message class instance of '" + messageClassName + "' from the message factory");
175                 this.messageQ = MessageQFactory.getInstance(messageClassName, options);
176             }
177         } catch (FactoryException e) {
178             final String error = "The message class cannot be instantiated. This is most likely a configuration"
179                     + " problem. Is the configured class available in the classpath?";
180             LOG.error(error, e);
181             throw new IllegalStateException(error, e);
182         }
183 
184         // initialize the validator object
185         try {
186             final String validatorClass = getOption(JaasConfigOptions.VALIDATOR_PASSWORD_CLASS);
187             if (validatorClass == null) {
188                 LOG.debug("Requesting default validator class from the validator factory");
189                 this.pwValidator = PasswordValidatorFactory.getInstance(options);
190             } else {
191                 LOG.debug("Requesting validator class instance of '" + validatorClass
192                         + "' from the validator factory");
193                 this.pwValidator = PasswordValidatorFactory.getInstance(validatorClass, options);
194             }
195         } catch (FactoryException e) {
196             final String error = "The validator class cannot be instantiated. This is most likely a configuration"
197                     + " problem. Is the configured class available in the classpath?";
198             LOG.error(error, e);
199             throw new IllegalStateException(error, e);
200         }
201 
202         // initialize the authenticator object
203         try {
204             final String authNticatorClass = getOption(JaasConfigOptions.AUTHENTICATOR_PASSWORD_CLASS);
205             if (authNticatorClass == null) {
206                 LOG.debug("Requesting default authenticator class from the authenticator factory");
207                 this.pwAuthenticator = PasswordAuthenticatorFactory.getInstance(options);
208             } else {
209                 LOG.debug("Requesting authenticator class instance of '" + authNticatorClass
210                         + "' from the authenticator factory");
211                 this.pwAuthenticator = PasswordAuthenticatorFactory.getInstance(authNticatorClass, options);
212             }
213         } catch (FactoryException e) {
214             final String error = "The validator class cannot be instantiated. This is most likely a configuration"
215                     + " problem. Is the configured class available in the classpath?";
216             LOG.error(error, e);
217             throw new IllegalStateException(error, e);
218         }
219 
220         LOG.info("Initialization complete");
221     }

223     /**
224      * {@inheritDoc}
225      *
226      * @return {@code true} if authentication succeeds
227      * @throws LoginException if this {@code LoginModule} is unable to perform the authentication. Catch a
228      *                        {@link javax.security.auth.login.FailedLoginException} to determine if the authentication
229      *                        failed (wrong username or password)
230      */
231     @Override
232     // Check is broken [LOG.info()]: PMD reports issues although log stmt is guarded. @todo revisit when upgrading PMD.
233     @SuppressWarnings("PMD.GuardLogStatementJavaUtil")
234     public final boolean login()
235             throws LoginException {
236 
237         LOG.debug("Attempting login");
238 
239         if (pCallbackHandler == null) {
240             final String error = "No CallbackHandler available to garner authentication information from the user";
241             LOG.error(error);
242             throw new LoginException(error);
243         }
244         Callback[] callbacks = new Callback[3];
245         callbacks[0] = new NameCallback("Username: ");
246         callbacks[1] = new PasswordCallback("Password: ", false);
247         callbacks[2] = new TextInputCallback("Domain: ");
248 
249         try {
250             pCallbackHandler.handle(callbacks);
251 
252             // store the username
253             username = ((NameCallback) callbacks[0]).getName();
254 
255             // store the password (i.e. a copy of the password)
256             final char[] tempPassword = ((PasswordCallback) callbacks[1]).getPassword();
257             password = tempPassword.clone();
258 
259             // clear the password in the callback
260             ((PasswordCallback) callbacks[1]).clearPassword();
261 
262             // store the domain
263             domain = ((TextInputCallback) callbacks[2]).getText();
264         } catch (java.io.IOException e) {
265             cleanState();
266             final String error = "Encountered an I/O exception during login";
267             LOG.warn(error, e);
268             throw Util.newLoginException(error, e);
269         } catch (UnsupportedCallbackException e) {
270             cleanState();
271             final String error =
272                     e.getCallback().toString() + " not available to garner authentication information from the user";
273             LOG.warn(error, e);
274             throw Util.newLoginException(error, e);
275         }
276 
277         LOG.debug("Attempting login - discovered user '" + username + "@" + domain + "'");
278 
279         // Using a try/catch construct for managing control flows is really a bad idea.
280         // Unfortunately, this is how JAAS works :-(
281         try {
282             // authenticate, and update state and pending subject if successful
283             pendingSubject = pwAuthenticator.authenticate(domain, username, password, pwValidator);
284 
285             // then clear the password
286             Util.zeroArray(password);
287 
288             try {
289                 audit.audit(Events.AUTHN_ATTEMPT, domain, username);
290                 messageQ.create(Events.AUTHN_ATTEMPT, domain, username);
291             } catch (AuditException | MessageQException e) {
292                 final String error = "Login successful for '" + username + "@" + domain
293                         + "', but cannot audit login attempt or create message event, and hence fail the operation";
294                 LOG.warn(error, e);
295                 throw Util.newLoginException(error, e);
296             }
297 
298             // string concatenation is only executed if log level is actually enabled
299             if (LOG.isInfoEnabled()) {
300                 LOG.info("Login complete for '" + username + "@" + domain + "'");
301             }
302             return true;
303         } catch (LoginException e) {
304             // the login failed
305 
306             // cache the username and domain, for they will be purged by "cleanState()"
307             final String tempUsername = username;
308             final String tempDomain = domain;
309 
310             cleanState();
311 
312             try {
313                 audit.audit(Events.AUTHN_FAILURE, tempDomain, tempUsername);
314                 messageQ.create(Events.AUTHN_FAILURE, tempDomain, tempUsername);
315             } catch (AuditException | MessageQException e2) {
316                 final String error = "Login failed for '" + tempUsername + "@" + tempDomain
317                         + "', but cannot audit login attempt or create message event";
318                 LOG.warn(error, e2);
319             }
320             final String error = "Login failed for '" + tempUsername + "@" + tempDomain + "'";
321             LOG.info(error, e);
322 
323             throw e;
324         }
325     }

327     /**
328      * {@inheritDoc}
329      */
330     @Override
331     // Check is broken [LOG.info()]: PMD reports issues although log stmt is guarded. @todo revisit when upgrading PMD.
332     @SuppressWarnings("PMD.GuardLogStatementJavaUtil")
333     public final boolean commit()
334             throws LoginException {
335 
336         LOG.debug("Committing authentication");
337 
338         if (pendingSubject == null) {
339             // the login method of this module has failed earlier, hence we do not need to clean up anything
340             // return 'false' to indicate that this module's login and/or commit method failed
341             // As the login method failed, the state of the module has already been cleared and we do not know
342             // the username / domain anymore. Hence no auditing / message queue notification, and not verbose logging.
343 
344             LOG.debug("Not committing authentication, as the authentication has failed earlier (login method)");
345 
346             return false;
347         } else {
348             // The login has succeeded!
349             // Store the principals from the pending subject both in the 'subject' object (because this is what JAAS
350             // will use later on), but also create a new 'committedSubject' object that this module uses to a) keep
351             // state and b) being able to remove the principals later
352 
353             LOG.debug("Committing authentication: '" + username + "@" + domain + "'");
354 
355             if (committedSubject == null) {
356                 committedSubject = new Subject();
357             } else {
358 
359                 // cache the username and domain, for they will be purged by "cleanState()"
360                 final String tempUsername = username;
361                 final String tempDomain = domain;
362 
363                 cleanState();
364 
365                 try {
366                     audit.audit(Events.AUTHN_ERROR, tempDomain, tempUsername);
367                     messageQ.create(Events.AUTHN_ERROR, tempDomain, tempUsername);
368                 } catch (AuditException | MessageQException e) {
369                     final String error = "Login post-processing failed for '" + tempUsername + "@" + tempDomain
370                             + "', but cannot audit login attempt or create message event";
371                     LOG.warn(error, e);
372                 }
373 
374                 final String error = "Expected the committed subject to be 'null' (yes, really <null>), but this was "
375                         + "not the case! Has the commit method been called multiple times on the same object instance?";
376                 LOG.warn(error);
377                 throw new LoginException(error);
378             }
379 
380             for (final Principal p : pendingSubject.getPrincipals()) {
381 
382                 // 1. Add the principals to the 'subject' object
383                 if (!pSubject.getPrincipals().contains(p)) {
384                     LOG.debug("Added principal " + p.getName() + " to subject");
385                     pSubject.getPrincipals().add(p);
386                 }
387 
388                 // 2. Add the principals to the 'committedSubject' object
389                 if (!committedSubject.getPrincipals().contains(p)) {
390                     LOG.debug("Added principal " + p.getName() + " to committed subject");
391                     committedSubject.getPrincipals().add(p);
392                 }
393             }
394 
395             try {
396                 audit.audit(Events.AUTHN_SUCCESS, domain, username);
397                 messageQ.create(Events.AUTHN_SUCCESS, domain, username);
398             } catch (AuditException | MessageQException e) {
399                 final String error = "Login succeeded for '" + username + "@" + domain
400                         + "', but cannot audit login success or create message event, and hence fail the operation";
401                 LOG.warn(error, e);
402                 throw Util.newLoginException(error, e);
403             }
404 
405             // string concatenation is only executed if log level is actually enabled
406             if (LOG.isInfoEnabled()) {
407                 LOG.info("Authentication committed for '" + username + "@" + domain + "'");
408             }
409 
410             // do not clean the state here, as we may still need it in case of an abort()
411             return true;
412         }
413     }

418     @Override
419     // Check is broken [LOG.info()]: PMD reports issues although log stmt is guarded. @todo revisit when upgrading PMD.
420     @SuppressWarnings("PMD.GuardLogStatementJavaUtil")
421     public final boolean abort()
422             throws LoginException {
423 
424         if (pendingSubject == null) {
425             // the login method of this module has failed earlier, hence we do not need to clean up anything
426             // return 'false' to indicate that this module's login and/or commit method failed
427             // As the login method failed, the state of the module has already been cleared and we do not know
428             // the username / domain anymore. Hence no auditing / message queue notification, and not verbose logging.
429 
430             LOG.debug("Aborting authentication, as the authentication has failed earlier (login method)");
431 
432             return false;
433         } else if (committedSubject == null) {
434             // the login method of this module succeeded, but the overall authentication failed
435 
436             // string concatenation is only executed if log level is actually enabled
437             if (LOG.isDebugEnabled()) {
438                 LOG.debug("Aborting authentication: '" + username + "@" + domain + "'");
439             }
440 
441             // cache the username and domain, for they will be purged by "cleanState()"
442             final String tempUsername = username;
443             final String tempDomain = domain;
444 
445             cleanState();
446 
447             try {
448                 audit.audit(Events.AUTHN_ABORT_COMMIT, tempDomain, tempUsername);
449                 messageQ.create(Events.AUTHN_ABORT_COMMIT, tempDomain, tempUsername);
450             } catch (AuditException | MessageQException e) {
451                 final String error = "Login post-processing failed after abort for '" + tempUsername + "@" + tempDomain
452                         + "', but cannot audit login attempt or create message event";
453                 LOG.warn(error, e);
454             }
455 
456             // string concatenation is only executed if log level is actually enabled
457             if (LOG.isInfoEnabled()) {
458                 LOG.info("Authentication aborted for '" + tempUsername + "@" + tempDomain + "'");
459             }
460             return true;
461         } else {
462             // overall authentication succeeded and commit succeeded, but someone else's commit failed
463             try {
464                 audit.audit(Events.AUTHN_ABORT_CHAIN, domain, username);
465                 messageQ.create(Events.AUTHN_ABORT_CHAIN, domain, username);
466             } catch (AuditException | MessageQException e) {
467                 final String error = "Login post-processing failed after abort for '" + username + "@" + domain
468                         + "', but cannot audit login attempt or create message event";
469                 LOG.warn(error, e);
470             }
471 
472             // cache the username and domain, for they will be purged by "logout()"
473             final String tempUsername = username;
474             final String tempDomain = domain;
475 
476             logout();
477 
478             // string concatenation is only executed if log level is actually enabled
479             if (LOG.isInfoEnabled()) {
480                 LOG.info("Authentication aborted for '" + tempUsername + "@" + tempDomain + "'");
481             }
482             return true;
483         }
484     }

489     @Override
490     // Check is broken [LOG.info()]: PMD reports issues although log stmt is guarded. @todo revisit when upgrading PMD.
491     @SuppressWarnings("PMD.GuardLogStatementJavaUtil")
492     public final boolean logout()
493             throws LoginException {
494 
495         final StringBuilder principals = new StringBuilder(":");
496 
497         // remove all the principals that we added in the commit() method from the 'subject' object
498         // (that's why we stored our principals in the 'committedSubject' object...)
499         if (committedSubject != null && committedSubject.getPrincipals() != null) {
500             for (final Principal p : committedSubject.getPrincipals()) {
501                 pSubject.getPrincipals().remove(p);
502 
503                 // string concatenation is only executed if log level is actually enabled
504                 if (LOG.isDebugEnabled()) {
505                     LOG.debug("Logging out subject: '" + p.getName() + "'");
506                 }
507 
508                 principals.append(p.getName()).append(':');
509 
510                 try {
511                     audit.audit(Events.AUTHN_LOGOUT, p.getName());
512                     messageQ.create(Events.AUTHN_LOGOUT, p.getName());
513                 } catch (AuditException | MessageQException e) {
514                     final String error = "Logout successful for '" + p.getName()
515                             + "', but cannot audit logout attempt or create message event";
516                     LOG.warn(error, e);
517                 }
518             }
519         }
520 
521         cleanState();
522 
523         // string concatenation is only executed if log level is actually enabled
524         if (LOG.isInfoEnabled()) {
525             LOG.info("Principals logged out: '" + principals + "'");
526         }
527 
528         return true;
529     }

531     /**
532      * Clean up any state associated with the current login attempt.
533      */
534     @SuppressWarnings("PMD.NullAssignment")
535     private void cleanState() {
536 
537         // null-assignments for de-referencing objects are okay
538         domain = null;
539         username = null;
540         Util.zeroArray(password);
541         pendingSubject = null;
542         committedSubject = null;
543     }
