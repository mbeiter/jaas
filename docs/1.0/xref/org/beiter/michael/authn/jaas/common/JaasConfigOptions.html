33  package org.beiter.michael.authn.jaas.common;
34  
35  /**
36   * The JAAS configuration options used by this library.
37   */
38  public enum JaasConfigOptions
39          implements ConfigOptions {
40  
41      //CHECKSTYLE:OFF
42      AUDIT_CLASS("jaas.audit.class"),
43      AUDIT_ENABLED("jaas.audit.enabled"),
44      MESSAGEQ_CLASS("jaas.messageq.class"),
45      MESSAGEQ_ENABLED("jaas.messageq.enabled"),
46      VALIDATOR_PASSWORD_CLASS("jaas.validator.password.class"),
47      AUTHENTICATOR_PASSWORD_CLASS("jaas.authenticator.password.class");
48  
49      private final String name;
50  
51      private JaasConfigOptions(final String keyName) {
52          this.name = keyName;
53      }
54  
55      public String getName() { return name; }
56      //CHECKSTYLE:ON
57  }
