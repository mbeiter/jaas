106     /**
107      * {@inheritDoc}
108      * <p/>
109      * See module documentation for a list of available options.
110      */
111     @Override
112     public final void init(final Map<String, ?> properties) {
113 
114         Validate.notNull(properties);
115 
116         // no need for defensive copies here, as all internal config values are calculated
117 
118         if (getOption(JdbcConfigOptions.JNDI_NAME, properties) != null
119                 && String.class.isInstance(getOption(JdbcConfigOptions.JNDI_NAME, properties))) {
120             jndiName.set((String) getOption(JdbcConfigOptions.JNDI_NAME, properties));
121         } else {
122             jndiName.set(null);
123         }
124 
125         // only init the connection driver and pool configuration if JNDI is not being used
126         if (jndiName.get() == null) {
127 
128             LOG.info("JNDI name is not configured, trying to create a connection pool instead");
129 
130             if (getOption(JdbcConfigOptions.POOL_DRIVER, properties) != null
131                     && String.class.isInstance(getOption(JdbcConfigOptions.POOL_DRIVER, properties))) {
132                 driver.set((String) getOption(JdbcConfigOptions.POOL_DRIVER, properties));
133             } else {
134                 driver.set(null);
135             }
136 
137             // the connection spec is cached, so that it has to be created only when init() is called
138             connectionSpec.set(ConfigUtil.getConnectionSpec(properties));
139 
140             // the connection pool spec is cached, so that it has to be created only when init() is called
141             connPoolSpec.set(ConfigUtil.getConnectionPoolSpec(properties));
142         }
143 
144         if (getOption(JdbcConfigOptions.SQL_USERQUERY, properties) != null
145                 && String.class.isInstance(getOption(JdbcConfigOptions.SQL_USERQUERY, properties))) {
146             sqlQuery.set((String) getOption(JdbcConfigOptions.SQL_USERQUERY, properties));
147         } else {
148             sqlQuery.set(null);
149         }
150     }

152     /**
153      * Authenticate a user by validating the user's password, returning a Subject with one or more {@code Principal}s
154      * set (if validation was successful), or throw a {@code LoginException} (if validation fails)
155      * <p/>
156      * This method connects to the database as configured in the the configuration provided during initialization, and
157      * uses the configured parameterized SQL query to retrieve a user's ID and credential using the provided userName
158      * and domain to locate them in the database. If initialization has not been completed, the authentication fails.
159      * <p/>
160      * The configured SQL query must take two query parameters, the first being the domain, the second being the
161      * userName. If the domain is not used in the database, the SQL query must be crafted so that the first parameter
162      * is irrelevant.
163      * <p/>
164      * The configured SQL query must return two columns, the first being the user's ID, the second being the credential
165      * against which the provided password is to be validated.
166      * <p/>
167      * If the JDBC call failed, no record could be found, the parameters do not match, or similar SQL issues ocured,
168      * a {@link javax.security.auth.login.LoginException} is thrown.
169      * <p/>
170      * The provided password is validated against the credential using the parent class' {@code passwordValidator}
171      * object.
172      * <p/>
173      * If the validation fails (i.e. the provided password does not match the credential), a
174      * {@link javax.security.auth.login.FailedLoginException} is thrown.
175      * <p/>
176      * If the validation is successful, a {@code Subject} is populated with three principals is returned: The
177      * user's ID as returned by the database, and both the user provided domain and the user provided principal (i.e.
178      * the the identifiers used to authenticate the users).
179      *
180      * @param domain            The white label domain in which the username is located
181      * @param userName          The username to authenticate with
182      * @param password          The password to authenticate with
183      * @param passwordValidator The validator to use for the password
184      * @return a callback for querying subject attributes.
185      * @throws LoginException when this {@code LoginModule} is unable to perform the authentication. Catch a
186      *                        {@link javax.security.auth.login.FailedLoginException} to determine if the authentication
187      *                        failed due to an incorrect password
188      */
189     // The logical flow in this method is not overly complex, the warning results from parameter validation
190     // CHECKSTYLE:OFF
191     @SuppressWarnings({"PMD.CyclomaticComplexity", "PMD.NPathComplexity"})
192     // CHECKSTYLE:ON
193     public final Subject authenticate(final String domain, final String userName, final char[] password,
194                                       final PasswordValidator passwordValidator)
195             throws LoginException {
196 
197         // make sure the credentials are not null
198         if (domain == null || userName == null || password == null) {
199             throw new LoginException("The credentials cannot be null");
200         }
201 
202         // SQL query is required
203         if (sqlQuery.get() == null || sqlQuery.get().length() == 0) {
204             final String error = "Invalid SQL user authentication query (query is null or empty)";
205             LOG.warn(error);
206             throw new LoginException(error);
207         }
208 
209         final UserRecord userRecord = getUserRecord(domain, userName);
210 
211         if (userRecord.getUserId() == null || userRecord.getUserId().length() == 0) {
212             final String error = "User ID for username '" + userName + "' is null or empty in the database";
213             LOG.warn(error);
214             throw new LoginException(error);
215         }
216 
217         if (userRecord.getCredential() == null || userRecord.getCredential().length() == 0) {
218             final String error = "Credential for username '" + userName + "' / user ID '" + userRecord.getUserId()
219                     + "' is null or empty in the database";
220             LOG.warn(error);
221             throw new LoginException(error);
222         }
223 
224         // no need for defensive copies of Strings, but create a defensive copy of the password
225         final char[] myPassword = password.clone();
226 
227         // convert the credential string to a char array
228         final char[] myCredential = userRecord.getCredential().toCharArray();
229 
230         if (!passwordValidator.validate(myPassword, myCredential)) {
231             final String error = "Invalid password for username '" + userName + "'";
232             LOG.info(error);
233             throw new FailedLoginException(error);
234         }
235 
236         // The authentication was successful!
237         // Create the subject and clean up confidential data as far as possible.
238 
239         // clear the char representation of the credential
240         Util.zeroArray(myCredential);
241 
242         // clear the defensive copy of the password created earlier
243         Util.zeroArray(myPassword);
244 
245         // create a principal that includes the username and domain name that were used to authenticate the user
246         final UserPrincipal userPrincipal = new UserPrincipal(userRecord.getUserId(), domain, userName);
247 
248         // wrap the principal in a Subject
249         final Subject subject = new Subject();
250         subject.getPrincipals().add(userPrincipal);
251 
252         return subject;
253     }

255     /**
256      * Retrieve a user record from the database, with the user record being uniquely identified through {@code domain}
257      * and {@code userName}.
258      *
259      * @param domain   The domain in which the {@code userName} is located
260      * @param userName The username to search for
261      * @return A user record containing domain, username, user ID (from the DB), and credentials (from the DB)
262      * @throws LoginException When the JDBC cvonnection failed, or the username / domain combination could not be found
263      */
264     // The SQL statement is retrieved from the configuration, and the admin is trusted
265     @edu.umd.cs.findbugs.annotations.SuppressWarnings("SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING")
266     private UserRecord getUserRecord(final String domain, final String userName)
267             throws LoginException {
268 
269         String userId;
270         String credential;
271 
272         Connection connection = null;
273         PreparedStatement statement = null;
274         ResultSet resultSet = null;
275 
276         try {
277             connection = getDatabaseConnection();
278 
279             statement = connection.prepareStatement(sqlQuery.get());
280             statement.setString(1, domain);
281             statement.setString(2, userName);
282 
283             resultSet = statement.executeQuery();
284 
285             if (resultSet.next()) {
286                 userId = resultSet.getString(1);
287                 credential = resultSet.getString(2);
288             } else {
289                 final String error = "Username '" + userName + "' does not exist (query returned zero results)";
290                 LOG.warn(error);
291                 throw new LoginException(error);
292             }
293 
294             resultSet.close();
295             statement.close();
296         } catch (SQLException e) {
297             final String error = "Error executing SQL query";
298             LOG.warn(error, e);
299             throw Util.newLoginException(error, e);
300         } finally {
301             DbUtil.close(resultSet);
302             DbUtil.close(statement);
303             DbUtil.close(connection);
304         }
305 
306         return new UserRecord(domain, userName, userId, credential);
307     }

309     /**
310      * Obtain a database connection - either a JNDI connection (directly from the factory), or a pooled JDBC connection
311      *
312      * @return a pooled database connection
313      * @throws LoginException when the connection cannot be retrieved from JNDI or the connection pool, or the pool
314      *                        cannot be created
315      */
316     private Connection getDatabaseConnection()
317             throws LoginException {
318 
319         Connection connection;
320         if (jndiName.get() != null && jndiName.get().length() > 0) {
321             try {
322                 connection = ConnectionFactory.getConnection(jndiName.get());
323             } catch (FactoryException e) {
324                 final String error = "Could not retrieve JNDI database connection";
325                 LOG.warn(error, e);
326                 throw Util.newLoginException(error, e);
327             }
328         } else {
329             try {
330                 // driver is required
331                 if (driver.get() == null || driver.get().length() == 0) {
332                     final String error = "Invalid database driver (driver name is null or empty)";
333                     LOG.warn(error);
334                     throw new FactoryException(error);
335                 }
336 
337                 if (connectionSpec.get() == null || connPoolSpec.get() == null) {
338                     final String error = "Database connection pool configuration has not been provided or initialized";
339                     LOG.warn(error);
340                     throw new FactoryException(error);
341                 }
342 
343                 connection = ConnectionFactory.getConnection(driver.get(), connectionSpec.get(), connPoolSpec.get());
344 
345             } catch (FactoryException e) {
346                 final String error = "Could not create pooled database connection";
347                 LOG.warn(error, e);
348                 throw Util.newLoginException(error, e);
349             }
350         }
351 
352         return connection;
353     }

355     /**
356      * Return the value of a JAAS configuration parameter.
357      *
358      * @param <T>        The type of the element
359      * @param key        The key to retrieve from the options
360      * @param properties The properties to retrieve values from
361      * @return The configuration value for the provided key
362      */
363     @SuppressWarnings("unchecked")
364     private static <T> T getOption(final ConfigOptions key, final Map<String, ?> properties) {
365 
366         // private method asserts
367         assert key != null : "The key cannot be null";
368 
369         return (T) properties.get(key.getName());
370     }
